import maya.cmds as cmds

def get_face_center(face_name):
    # Get the vertices of the face
    vertices = cmds.polyListComponentConversion(face_name, toVertex=True)
    vertices = cmds.filterExpand(vertices, selectionMask=31)  # Filter for vertices

    # Calculate the average position of all vertices to find the centroid
    x, y, z = 0.0, 0.0, 0.0
    for vertex in vertices:
        position = cmds.pointPosition(vertex, world=True)
        x += position[0]
        y += position[1]
        z += position[2]

    # Average out the sum of positions
    centroid = (x / len(vertices), y / len(vertices), z / len(vertices))
    return centroid

# Usage example:
face_name = 'yourMesh.f[faceIndex]'  # Replace with your mesh name and face index
center = get_face_center(face_name)
print("Center of the face is:", center)
