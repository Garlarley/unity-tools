import maya.cmds as cmds

def soften_mesh_edges(mesh_name, angle=180):
    """
    Soften the edges of the given mesh.

    :param mesh_name: str, the name of the mesh whose edges will be softened
    :param angle: float, the angle threshold to soften the edges
    """
    # Select all edges of the mesh
    cmds.select(f'{mesh_name}.e[*]', r=True)
    
    # Soften the edges
    cmds.polySoftEdge(angle=angle, constructionHistory=False)

    # Clear the selection
    cmds.select(clear=True)

# Example usage:
mesh_name = 'pCube1'  # Replace with the name of your mesh
soften_mesh_edges(mesh_name)
