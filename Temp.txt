import maya.api.OpenMaya as OpenMaya

def get_bounding_box(mesh_name):
    # Get the selection list
    selection_list = OpenMaya.MSelectionList()
    selection_list.add(mesh_name)
    
    # Get the MDagPath object for the mesh
    dag_path = selection_list.getDagPath(0)
    dag_path.extendToShape()
    
    # Get the MFnDagNode function set for the mesh
    dag_node = OpenMaya.MFnDagNode(dag_path)
    
    # Get the bounding box
    bounding_box = dag_node.boundingBox
    
    # Get the min and max points of the bounding box
    min_point = bounding_box.min
    max_point = bounding_box.max
    
    # Return the bounding box as a dictionary
    bbox = {
        'minX': min_point.x, 'minY': min_point.y, 'minZ': min_point.z,
        'maxX': max_point.x, 'maxY': max_point.y, 'maxZ': max_point.z
    }
    
    return bbox

# Example usage
mesh_name = 'pCube1'  # Replace with your actual mesh name
bbox = get_bounding_box(mesh_name)
print("Bounding Box:", bbox)
