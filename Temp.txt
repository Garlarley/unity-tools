import maya.api.OpenMaya as OpenMaya

def get_face_center(face_name):
    # Parse the face name to get the mesh name and face index
    split_name = face_name.split('.')
    mesh_name = split_name[0]
    face_index = int(split_name[1][2:-1])  # Assumes face_name is like 'pCube1.f[0]'

    # Get the MObject for the mesh
    selectionList = OpenMaya.MSelectionList()
    selectionList.add(mesh_name)
    mesh_mobject = selectionList.getDependNode(0)
    mesh_fn = OpenMaya.MFnMesh(mesh_mobject)

    # Get the face's vertices
    face_iter = OpenMaya.MItMeshPolygon(mesh_mobject)
    face_iter.setIndex(face_index)
    points = OpenMaya.MPointArray()
    face_iter.getPoints(points, OpenMaya.MSpace.kWorld)

    # Calculate the centroid
    centroid = OpenMaya.MPoint()
    num_points = points.length()
    for i in range(num_points):
        centroid += points[i]
    centroid /= num_points

    return (centroid.x, centroid.y, centroid.z)

# Example usage
face_name = 'pCube1.f[0]'  # Replace with your actual face name
centroid = get_face_center(face_name)
print("Centroid of the face is:", centroid)
