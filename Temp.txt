import maya.cmds as cmds
import time  # Only for demonstration purposes (e.g., simulating a long task)

# Create a progress bar window
if cmds.window('progressBarWindow', exists=True):
    cmds.deleteUI('progressBarWindow', window=True)

cmds.window('progressBarWindow', title='Progress Bar Example')
cmds.columnLayout()
progress_control = cmds.progressBar(maxValue=100, width=300)

cmds.showWindow('progressBarWindow')

# Simulate a long process with a loop
for i in range(100):
    # Here you'd put your actual processing code
    time.sleep(0.1)  # Simulate time-consuming task

    # Update the progress bar
    cmds.progressBar(progress_control, edit=True, step=1)

# Once the process is complete, you can delete the progress bar window
cmds.deleteUI('progressBarWindow', window=True)
