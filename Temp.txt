import maya.cmds as cmds

# Replace 'extractedMesh' with the name of your extracted mesh
extractedMesh = 'extractedMesh'
# Replace 'originalMesh' with the name of your original terrain mesh
originalMesh = 'originalMesh'

# Unlock the normals on both meshes
cmds.polyNormalPerVertex(extractedMesh, unFreezeNormal=True)
cmds.polyNormalPerVertex(originalMesh, unFreezeNormal=True)

# Soften the edge normals on both meshes
cmds.polySoftEdge(extractedMesh, angle=180)
cmds.polySoftEdge(originalMesh, angle=180)

# Average normals along the border edges if necessary
# This step may require you to specify the border edges explicitly
borderEdges = cmds.polyListComponentConversion(
    [extractedMesh, originalMesh], 
    fromFace=True, 
    toEdge=True, 
    border=True
)
borderEdges = cmds.filterExpand(borderEdges, selectionMask=32, expand=True)
if borderEdges:
    cmds.polyAverageNormal(borderEdges, averageNormalType=0, averageNormalArea=2)

# Merge vertices if they are separate
# This step assumes the vertices are close enough to merge without altering the mesh form
cmds.select([extractedMesh, originalMesh])
cmds.polyMergeVertex(d=0.01, am=1, ch=True)
